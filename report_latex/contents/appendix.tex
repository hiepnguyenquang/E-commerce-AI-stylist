\section{Lệnh chạy hệ thống local}

\begin{enumerate}
    \item Cài đặt dependency Node.js bằng lệnh \code{pnpm install}.
    \item Khởi động các thành phần hạ tầng trong Docker bằng \code{docker compose up -d}.
    \item Khi PostgreSQL, Redis và RabbitMQ đã sẵn sàng, chạy migration Medusa bằng \code{pnpm -F @fundamental/api-medusa exec medusa db:migrate}.
    \item Khởi chạy frontend và Medusa từ thư mục gốc bằng \code{pnpm dev}.
    \item Chạy AI service trong thư mục \code{apps/ai-service} bằng \code{uv run uvicorn main:app --reload}.
\end{enumerate}

\section{Checklist minh chứng cần bổ sung}

\begin{itemize}
    \item Ảnh giao diện trang chủ, trang sản phẩm, AI Stylist, AI Profile, Wardrobe và checkout.
    \item Ảnh trước/sau VTON cho local engine và cloud engine nếu môi trường chạy cho phép.
    \item Log RabbitMQ cho quá trình đồng bộ sản phẩm, đồng bộ tủ đồ và xử lý tác vụ VTON.
    \item Log hoặc ảnh chứng minh frontend nhận event SSE \code{vision_update}.
    \item Kết quả thử AI Stylist với truy vấn hợp lệ và truy vấn ngoài miền để chứng minh guardrail.
    \item Bảng latency sau khi đo trên cấu hình máy cụ thể.
\end{itemize}
